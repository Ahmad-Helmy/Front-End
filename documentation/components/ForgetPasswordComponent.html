import { Component, OnInit } from "@angular/core";

@Component({
  selector: "app-forget-password",
  templateUrl: "./forget-password.component.html",
  styleUrls: ["./forget-password.component.css"]
})
export class ForgetPasswordComponent implements OnInit {
  Email: string;
  reset = false;

  constructor() {}

  ngOnInit() {}
}

<div class="m-auto text-center w-75">
    <h1 class="m-4">Password Reset</h1>
    <div *ngIf='!reset;else sended'>
        <h4 class="my-5">
            Enter your <b>Spotify username</b>, or the
            <b>email address</b> that you used to register. We'll send you an email with your
            username and a link to reset your password.
        </h4>

        <mat-form-field class="w-75">
            <mat-label>Email address or Username</mat-label>
            <input matInput type="text" [(ngModel)]='Email' name="email" #email="ngModel" required>
            <mat-error *ngIf="email.errors">
                <span *ngIf="email.errors.required">Please enter your email.</span>
            </mat-error>
        </mat-form-field>
        <button class="my-4 custom-btn btn btn-green text-white w-25">Send</button>
